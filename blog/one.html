# 一、存储桶

# 接口前缀

http://warehouse.ylmc2018.cn

# 1.1 查询所有存储桶信息

# 接口地址

/api/WStorageBucket

# 无参数,get请求

# 展示

[{
	"sb_id": 36,
	"sb_name": "北边仓库",
	"sb_capability": 10000,
	"sb_existence": 0,
	"sb_tablename": "w_goods_8c69eb6",
	"sb_remarks": "阿道夫",
	"sb_flag": 0
}

# 1.2 添加存储桶，添加仓库

# 接口地址

/api/WStorageBucket

# 请求方式：POST

参数名称 参数类型 说明
sb_name string 存储桶名称
sb_capability int 存储桶容量
sb_existence int 存储桶已使用
sb_remarks string 备注说明

# 插入成功后显示code=0

# 1.3 删除仓库

# 接口地址

/api/WStorageBucket

# 请求方式：Delete

# 说明

# 参数为id地址传递，例如要删除id为40的仓库传递地址为

/api/WStorageBucket/40

# 删除成功后显示code=0

# 1.4 修改仓库信息

# 接口地址

/api/WStorageBucket

# 请求方式：Put

参数名称 参数类型 说明
sb_id int 要修改的存储桶id
sb_name string 修改后存储桶名称
sb_capability int 修改后存储桶容量
sb_existence int 修改后存储桶使用量
sb_remarks string 修改后的备注

# 修改成功后显示code=0
